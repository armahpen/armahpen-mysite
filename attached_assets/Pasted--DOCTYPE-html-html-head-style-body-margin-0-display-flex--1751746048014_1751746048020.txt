  <script>
    const planets = document.querySelectorAll('.planet');
    let angle = 0;

    function animate() {
      const centerX = 400;
      const centerY = 300;
      const orbits = [
        { rx: 100, ry: 50 },
        { rx: 150, ry: 75 },
        { rx: 200, ry: 100 },
        { rx: 250, ry: 125 },
        { rx: 300, ry: 150 }
      ];

      planets.forEach((planet, index) => {
        const orbit = orbits[index];
        const x = centerX + orbit.rx * Math.cos(angle + index * Math.PI / 2);
        const y = centerY + orbit.ry * Math.sin(angle + index * Math.PI / 2);
        planet.setAttribute('cx', x);
        planet.setAttribute('cy', y);
      });

      angle += 0.02;
      requestAnimationFrame(animate);
    }

    animate();
  </script>
